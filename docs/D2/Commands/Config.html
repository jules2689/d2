def call(args, _name)
  case args.shift
  when 'set'
    set(args)
  when 'unset'
    unset(args)
  when 'get'
    get(args)
  when 'list', nil
    logger.info D2::Config.to_s
  else
    logger.info "Unrecognized command. Please see usage\n#{self.class.help}"
  end
end
